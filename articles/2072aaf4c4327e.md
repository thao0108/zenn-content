## for
```js
function sum(array) {
  // 配列が渡されてくるので、ここで順番に処理してください。  
  let sum = 0;
  for (let i = 0; i < array.length; i++) {
    sum += array[i];
  }
  return sum;
}
```

## reduce
```js
function sum(array) {
  // 配列が渡されてくるので、ここで順番に処理してください。  
  return array.reduce((a, b) => {
    return a + b;
  }, 0);
}
```
